117  bool gaussianElimination(Matrix M, double& result)
118  {
119  size_t numberOfSwaps{};
120  /*
121  transform matrix M into upper triangular form
122  */
123  for (size_t i = 0; i < M.size(); ++i)
124  {
125  // if the leading value (i-th column) is zero, try swaping its row with some row below it
126  if (isZero(M[i][i]))
127  {
128  // if no appropriate row was found -> report failure (false)
129  if (!swapRows(M, i, i))
130  return false;
131  // count number of row swaps
132  else
133  numberOfSwaps++;
134  }
135 
136  /*
137  transform (through elementary matrix operations) every row below i-th row,
138  so that every value in i-th column is equal to zero
139  */
140  for (size_t j = i + 1; j < M.size(); ++j)
141  {
142 
143  double multiplyFactor = M[j][i] / M[i][i];
144  size_t column = i;
145 
146  // subtract i-th row (multiplied by multiplyFactor) from j-th row
147  std::transform
148  (M[j].begin() + i, M[j].end(), // source
149  M[j].begin() + i, // destination
150  [&](double value) // operation
151  {return value - multiplyFactor*M[i][column++]; });
152  }
153  }
154 
155  // determinant of a matrix in triangular form is a product of every element on that matrix's diagonal
156  double tmp {1};
157  for (size_t i = 0; i < M.size(); ++i)
158  {
159  tmp *= M[i][i];
160  }
161  // check how many times rows were swapped
162  // odd - change determinant's sign
163  // even - don't change determinant's sign
164  if ((numberOfSwaps % 2) != 0) // odd number of times
165  tmp *= -1;
166 
167 
168  result = tmp;
169  return true;
170  }
